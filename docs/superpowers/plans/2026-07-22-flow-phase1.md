# Flow Phase 1 (MVP) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Node-canvas "Flow" feature: chain saved REST/gRPC requests (snapshots) with delays, run them sequentially with `{{steps.*}}` data passing, manage/export flows in a "Saved Flows" tab.

**Architecture:** Flows are self-contained JSON files in `~/RequestsMin/flows/` (Rust CRUD mirrors `req_*`, flow schema is opaque `serde_json::Value` to Rust). Editor is a React Flow canvas inside a new multi-instance tab kind `"flow"`; state mirrors the existing `requestTabs` pattern (`flowTabs` in the zustand store). The engine is frontend TS (mirrors `runRequest.ts`): it resolves `{{steps.*}}`/`{{vars.*}}` tokens itself and leaves `{{envVar}}` tokens for the existing Rust interpolation — zero backend execution changes.

**Tech Stack:** Tauri 2 + React 18 + Zustand + TypeScript. One new dependency: `@xyflow/react` (React Flow 12). Tests: `node --test` for `.test.mjs` (they import `.ts` directly, see `src/lib/collectionDrop.test.mjs`), `cargo test` in `src-tauri`.

Spec: `docs/superpowers/specs/2026-07-22-flow-design.md` — read it first.

## Global Constraints

- **Commits are HUMAN-ONLY** (user global CLAUDE.md). Never run `git commit`/`git push`. Each task ends with a "ready for commit" checkpoint: report the suggested message and stop staging nothing.
- Only new dependency allowed: `@xyflow/react`. Nothing else (no dnd, no animation, no JSONPath libs).
- Code style: match repo — 2-space indent, double quotes, terse comments only for non-obvious constraints.
- Typecheck gate for every frontend task: `npx tsc --noEmit` from repo root must pass.
- Rust gate: `cargo test` inside `src-tauri/` must pass.
- Snapshot rule: a request dropped into a flow is `structuredClone`d; env vars and gRPC proto sources stay live references (spec §1/§9).
- WebSocket requests are rejected from flows (validation error + drop rejection).
- Step failure semantics: HTTP status ≥ 400 or gRPC `statusCode !== "OK"` ⇒ step `failed`; flow stops (marks the rest `skipped`) unless that node's `onError === "continue"`.
- `{{steps.…}}`/`{{vars.…}}` resolved in frontend; ALL other `{{…}}` tokens must reach the backend untouched.

fn flows_dir(root: &Path) -> PathBuf { root.join("flows") }

/// ids are FE-generated slugs; reject anything that could traverse paths
fn flow_path(root: &Path, id: &str) -> Result<PathBuf, String> {
    if id.is_empty() || !id.chars().all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_') {
        return Err(format!("invalid flow id: {id}"));
    }
    Ok(flows_dir(root).join(format!("{id}.json")))
}

#[derive(serde::Serialize)]
#[serde(rename_all = "camelCase")]
pub struct FlowMeta { pub id: String, pub name: String, pub node_count: usize }

pub fn list_flows(root: &Path) -> Vec<FlowMeta> {
    let Ok(entries) = std::fs::read_dir(flows_dir(root)) else { return vec![] };
    let mut flows: Vec<FlowMeta> = entries
        .filter_map(|e| {
            let path = e.ok()?.path();
            if path.extension()? != "json" { return None; }
            let v: serde_json::Value = serde_json::from_str(&std::fs::read_to_string(&path).ok()?).ok()?;
            Some(FlowMeta {
                id: v.get("id")?.as_str()?.to_string(),
                name: v.get("name").and_then(|n| n.as_str()).unwrap_or("(unnamed)").to_string(),
                node_count: v.get("nodes").and_then(|n| n.as_array()).map_or(0, |a| a.len()),
            })
        })
        .collect();
    flows.sort_by(|a, b| a.name.to_lowercase().cmp(&b.name.to_lowercase()));
    flows
}

#[tauri::command]
pub fn flow_list() -> Vec<FlowMeta> { list_flows(&root_dir()) }

```ts
import type { Flow, FlowEdge, FlowNode } from "./types";
import { isRequestNode } from "./types";

export interface FlowIssue { level: "error" | "warn"; message: string; nodeId?: string }

/** Kahn topological sort; queue seeded in node order so execution is deterministic. Null on cycle. */
export function topoOrder(nodes: readonly FlowNode[], edges: readonly FlowEdge[]): string[] | null {
  const ids = new Set(nodes.map((n) => n.id));
  const indegree = new Map<string, number>(nodes.map((n) => [n.id, 0]));
  const out = new Map<string, string[]>();
  for (const e of edges) {
    if (!ids.has(e.source) || !ids.has(e.target)) continue;
    indegree.set(e.target, (indegree.get(e.target) ?? 0) + 1);
    out.set(e.source, [...(out.get(e.source) ?? []), e.target]);
  }
  const queue = nodes.filter((n) => indegree.get(n.id) === 0).map((n) => n.id);
  const order: string[] = [];
  while (queue.length) {
    const id = queue.shift()!;
    order.push(id);
    for (const next of out.get(id) ?? []) {
      const d = indegree.get(next)! - 1;
      indegree.set(next, d);
      if (d === 0) queue.push(next);
    }
  }
  return order.length === nodes.length ? order : null;
}

export function validateFlow(flow: Flow): FlowIssue[] {
  const issues: FlowIssue[] = [];
  if (!flow.nodes.length) issues.push({ level: "error", message: "Flow has no nodes" });
  if (flow.nodes.length && topoOrder(flow.nodes, flow.edges) === null) issues.push({ level: "error", message: "Flow has a cycle" });
  const counts = new Map<string, number>();
  for (const n of flow.nodes) counts.set(n.key, (counts.get(n.key) ?? 0) + 1);
  for (const [key, count] of counts) if (count > 1) issues.push({ level: "error", message: `Duplicate step key "${key}"` });
  const ids = new Set(flow.nodes.map((n) => n.id));
  for (const e of flow.edges) if (!ids.has(e.source) || !ids.has(e.target)) issues.push({ level: "warn", message: "Edge references a missing node" });
  for (const n of flow.nodes) {
    if (isRequestNode(n) && n.config.request.protocol === "ws") {
      issues.push({ level: "error", nodeId: n.id, message: `Step "${n.key}": WebSocket requests are not supported in flows` });
    }
  }
  const linked = new Set(flow.edges.flatMap((e) => [e.source, e.target]));
  if (flow.nodes.length > 1) {
    for (const n of flow.nodes) if (!linked.has(n.id)) issues.push({ level: "warn", nodeId: n.id, message: `Step "${n.key}" is not connected` });
  }
  return issues;
}

/** Unique step key from a request name: "Login user" -> "login-user", suffixing -2, -3… on collision. */
export function stepKeyFor(name: string, taken: ReadonlySet<string>): string {
  const base = name.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-|-$/g, "") || "step";
  let key = base;
  for (let n = 2; taken.has(key); n++) key = `${base}-${n}`;
  return key;
}
```

```ts
import type { KV, Request } from "../api";
import type { Flow, FlowRun } from "./types";
import { isRequestNode } from "./types";

export interface StepRefCtx { steps: Record<string, unknown>; vars: Record<string, unknown> }

const kvRecord = (kvs: KV[]): Record<string, string> => Object.fromEntries(kvs.map((kv) => [kv.key.toLowerCase(), kv.value]));
const tryParse = (text: string): unknown => { try { return JSON.parse(text); } catch { return undefined; } };

/** Context of finished steps: steps.<key>.{request, response.{status, headers, body, bodyText, timeMs}}. */
export function buildStepCtx(flow: Flow, run: FlowRun): StepRefCtx {
  const steps: Record<string, unknown> = {};
  for (const node of flow.nodes) {
    const result = run.steps[node.id];
    if (!result || result.status !== "success" || !isRequestNode(node)) continue;
    const r = result.response;
    let response: unknown;
    if (r && "status" in r) {
      response = { status: r.status, headers: kvRecord(r.headers), body: tryParse(r.body), bodyText: r.body, timeMs: r.timeMs };
    } else if (r) {
      response = { status: r.statusCode, headers: kvRecord(r.headers), trailers: kvRecord(r.trailers), body: tryParse(r.bodyJson), bodyText: r.bodyJson, timeMs: r.timeMs };
    }
    steps[node.key] = { request: result.resolvedRequest, response };
  }
  return { steps, vars: {} };
}

export function resolvePath(root: unknown, path: string): unknown {
  let cur = root;
  for (const part of path.split(".")) {
    if (cur == null || typeof cur !== "object") return undefined;
    cur = (cur as Record<string, unknown>)[part];
  }
  return cur;
}

/** Replace {{steps.*}} / {{vars.*}} tokens only; every other {{…}} is left for backend env interpolation. */
export function substituteRefs(text: string, ctx: StepRefCtx): string {
  return text.replace(/\{\{\s*((?:steps|vars)\.[^{}]+?)\s*\}\}/g, (_, ref: string) => {
    const value = resolvePath(ctx, ref);
    if (value === undefined) throw new Error(`Cannot resolve {{${ref}}} — the step has not run yet, or the path is wrong`);
    return typeof value === "string" ? value : JSON.stringify(value);
  });
}

/** Deep-copy `request`, substituting step refs in every string field. */
export function substituteRequest(request: Request, ctx: StepRefCtx): Request {
  const walk = (v: unknown): unknown => {
    if (typeof v === "string") return substituteRefs(v, ctx);
    if (Array.isArray(v)) return v.map(walk);
    if (v && typeof v === "object") return Object.fromEntries(Object.entries(v).map(([k, val]) => [k, walk(val)]));
    return v;
  };
  return walk(structuredClone(request)) as Request;
}
```

export async function saveActiveFlow() {
  const s = useApp.getState();
  const tab = s.tabs.find((t) => t.id === s.activeTabId);
  if (!tab || tab.kind !== "flow") return;
  const ft = s.flowTabs[tab.id];
  if (!ft) return;
  await api.flowWrite(ft.flowId, ft.flow);
  s.updateFlowTab(tab.id, { original: JSON.stringify(ft.flow) });
  s.showToast("Saved", `${ft.flow.name} written to disk.`);
}

export async function renameFlow(id: string, name: string) {
  const flow = await api.flowRead(id);
  await api.flowWrite(id, { ...flow, name });
  const s = useApp.getState();
  for (const tab of s.tabs) {
    const ft = s.flowTabs[tab.id];
    if (ft?.flowId !== id) continue;
    s.updateFlowTab(tab.id, {
      flow: { ...ft.flow, name },
      original: JSON.stringify({ ...JSON.parse(ft.original), name }),
    });
  }
}

export async function duplicateFlow(id: string) {
  const flow = await api.flowRead(id);
  const copy = { ...flow, id: newFlowId(), name: `${flow.name} copy` };
  await api.flowWrite(copy.id, copy);
}

export async function deleteFlow(id: string) {
  const s = useApp.getState();
  for (const tab of [...s.tabs]) if (s.flowTabs[tab.id]?.flowId === id) s.closeTab(tab.id);
  await api.flowDelete(id);
}

  useApp.getState().showToast("Exported", dest);

function Canvas({ tabId, onRunNode }: { tabId: string; onRunNode?: (nodeId: string) => void }) {
  const ft = useApp((s) => s.flowTabs[tabId]);
  const updateFlowTab = useApp((s) => s.updateFlowTab);
  const showToast = useApp((s) => s.showToast);
  const theme = useApp((s) => s.theme);
  const { screenToFlowPosition } = useReactFlow();

  const rfNodes: Node[] = useMemo(() => ft.flow.nodes.map((n) => ({
    id: n.id, type: n.type, position: n.position, selected: ft.selectedNodeId === n.id,
    data: { node: n, status: ft.run?.steps[n.id]?.status ?? "idle", onRun: onRunNode },
  })), [ft.flow.nodes, ft.run, ft.selectedNodeId, onRunNode]);

  const rfEdges: Edge[] = useMemo(() => ft.flow.edges.map((e) => ({
    ...e,
    animated: ft.run?.steps[e.source]?.status === "success" && ft.run?.steps[e.target]?.status === "running",
  })), [ft.flow.edges, ft.run]);

  const setFlow = (nodes: FlowNode[], edges = ft.flow.edges) =>
    updateFlowTab(tabId, { flow: { ...ft.flow, nodes, edges } });

  const onNodesChange = useCallback((changes: NodeChange[]) => {
    const applied = applyNodeChanges(changes, rfNodes);
    const byId = new Map(applied.map((n) => [n.id, n]));
    const selection = changes.find((c) => c.type === "select");
    const nodes = ft.flow.nodes
      .filter((n) => byId.has(n.id)) // removals
      .map((n) => ({ ...n, position: byId.get(n.id)!.position }));
    updateFlowTab(tabId, {
      flow: { ...ft.flow, nodes, edges: ft.flow.edges.filter((e) => byId.has(e.source) && byId.has(e.target)) },
      ...(selection ? { selectedNodeId: selection.selected ? selection.id : null } : {}),
    });
  }, [ft, rfNodes, tabId, updateFlowTab]);

  const onEdgesChange = useCallback((changes: EdgeChange[]) => {
    setFlow(ft.flow.nodes, applyEdgeChanges(changes, rfEdges).map(({ id, source, target, sourceHandle }) =>
      ({ id, source, target, sourceHandle: sourceHandle ?? undefined })));
  }, [ft, rfEdges]);

  const onConnect = useCallback((conn: Connection) => {
    setFlow(ft.flow.nodes, addEdge({ ...conn, id: `e-${conn.source}-${conn.target}-${Date.now().toString(36)}` }, rfEdges)
      .map(({ id, source, target, sourceHandle }) => ({ id, source, target, sourceHandle: sourceHandle ?? undefined })));
  }, [ft, rfEdges]);

  const onDrop = useCallback(async (event: React.DragEvent) => {
    event.preventDefault();
    try {
      const item = JSON.parse(event.dataTransfer.getData("application/json"));
      if (item.kind !== "request") return;
      const request = structuredClone(await api.reqRead(item.collectionId, item.relPath));
      if (request.protocol === "ws") { showToast("Not supported", "WebSocket requests can't be used in flows.", "warn"); return; }
      const cur = useApp.getState().flowTabs[tabId];
      const node: FlowNode = {
        id: nextNodeId(),
        key: stepKeyFor(request.name, new Set(cur.flow.nodes.map((n) => n.key))),
        type: "request",
        position: screenToFlowPosition({ x: event.clientX, y: event.clientY }),
        config: { request, origin: { collectionId: item.collectionId, relPath: item.relPath } },
      };
      useApp.getState().updateFlowTab(tabId, { flow: { ...cur.flow, nodes: [...cur.flow.nodes, node] }, selectedNodeId: node.id });
    } catch { /* not a request payload */ }
  }, [screenToFlowPosition, showToast, tabId]);

  return (
    <ReactFlow
      nodes={rfNodes} edges={rfEdges} nodeTypes={nodeTypes}
      onNodesChange={onNodesChange} onEdgesChange={onEdgesChange} onConnect={onConnect}
      onDrop={(e) => void onDrop(e)} onDragOver={(e) => { e.preventDefault(); e.dataTransfer.dropEffect = "copy"; }}
      colorMode={themeBase(theme) === "light" ? "light" : "dark"}
      fitView deleteKeyCode={["Backspace", "Delete"]} proOptions={{ hideAttribution: true }}
    >
      <Background gap={16} />
      <Controls showInteractive={false} />
    </ReactFlow>
  );
}

```ts
import { api } from "../api";
import { useApp } from "../../store";
import { buildStepCtx, substituteRequest } from "./stepRefs";
import { topoOrder, validateFlow } from "./validate";
import { isRequestNode, type DelayNodeConfig, type FlowRun } from "./types";

const cancels = new Map<string, { cancelled: boolean }>();

export function cancelFlow(tabId: string) {
  const flag = cancels.get(tabId);
  if (flag) flag.cancelled = true;
}

const sleep = async (ms: number, flag: { cancelled: boolean }) => {
  const end = Date.now() + ms;
  while (Date.now() < end && !flag.cancelled) await new Promise((r) => setTimeout(r, Math.min(100, end - Date.now())));
};

export async function runActiveFlow() {
  const s = useApp.getState();
  const tab = s.tabs.find((t) => t.id === s.activeTabId);
  if (tab?.kind === "flow") await runFlow(tab.id);
}

export async function runFlow(tabId: string, onlyNodeId?: string) {
  const s = useApp.getState();
  const ft = s.flowTabs[tabId];
  if (!ft || ft.running) return;
  const flow = ft.flow;
  const errors = validateFlow(flow).filter((i) => i.level === "error");
  if (errors.length) { s.showToast("Flow invalid", errors.map((i) => i.message).join(" · "), "err"); return; }
  const order = topoOrder(flow.nodes, flow.edges)!;
  const targets = onlyNodeId ? [onlyNodeId] : order;
  const flag = { cancelled: false };
  cancels.set(tabId, flag);

  // single-node rerun keeps the previous run's results as upstream context
  const steps: FlowRun["steps"] = onlyNodeId ? structuredClone(ft.run?.steps ?? {}) : {};
  if (!onlyNodeId) for (const n of flow.nodes) steps[n.id] = { status: "idle" };
  const run: FlowRun = { startedAt: Date.now(), status: "running", steps };
  const env = s.activeEnv;
  const push = (running: boolean) => useApp.getState().updateFlowTab(tabId, { run: structuredClone(run), running });
  push(true);

  const t0 = Date.now();
  let stopped = false;
  for (const nodeId of targets) {
    const node = flow.nodes.find((n) => n.id === nodeId)!;
    if (flag.cancelled || stopped) { run.steps[nodeId] = { status: "skipped" }; push(true); continue; }
    run.steps[nodeId] = { status: "running" };
    push(true);
    const st = Date.now();
    try {
      if (node.type === "delay") {
        await sleep((node.config as DelayNodeConfig).ms, flag);
        run.steps[nodeId] = { status: flag.cancelled ? "skipped" : "success", timeMs: Date.now() - st };
      } else if (isRequestNode(node)) {
        const resolved = substituteRequest(node.config.request, buildStepCtx(flow, run));
        const response = resolved.protocol === "grpc"
          ? await api.grpcUnary(env, resolved.grpc!)
          : await api.httpRequest(env, resolved);
        const ok = "status" in response ? response.status < 400 : response.statusCode === "OK";
        run.steps[nodeId] = {
          status: ok ? "success" : "failed", timeMs: Date.now() - st, resolvedRequest: resolved, response,
          error: ok ? undefined : `step returned ${"status" in response ? response.status : response.statusCode}`,
        };
        if (!ok && node.config.onError !== "continue") stopped = true;
        // mirror the response into the node editor pane if it's open
        const editorId = `flowreq:${tabId}:${nodeId}`;
        if (useApp.getState().requestTabs[editorId]) useApp.getState().updateRequestTab(editorId, { response });
      }
    } catch (err) {
      run.steps[nodeId] = { status: "failed", timeMs: Date.now() - st, error: String(err) };
      stopped = true;
    }
    push(true);
  }
  run.totalMs = Date.now() - t0;
  run.status = flag.cancelled ? "cancelled" : Object.values(run.steps).some((r) => r.status === "failed") ? "failed" : "success";
  cancels.delete(tabId);
  push(false);
  if (!onlyNodeId) {
    const s2 = useApp.getState();
    s2.showToast(run.status === "success" ? "Flow finished" : `Flow ${run.status}`, `${run.totalMs} ms`, run.status === "success" ? "ok" : run.status === "failed" ? "err" : "warn");
  }
}
```

Behavior notes (already spec'd): cancel takes effect **between** steps — an in-flight backend request cannot be aborted in Phase 1; per-step timeout/retry are Phase 2.

import { useApp } from "../../store";
import { topoOrder } from "../../lib/flow/validate";
import { isRequestNode } from "../../lib/flow/types";
import { Icon } from "../../ui/Icon";

export function RunReport({ tabId }: { tabId: string }) {
  const ft = useApp((s) => s.flowTabs[tabId]);
  const updateFlowTab = useApp((s) => s.updateFlowTab);
  if (!ft?.run) return null;
  const { run, flow } = ft;
  const order = topoOrder(flow.nodes, flow.edges) ?? flow.nodes.map((n) => n.id);
  const nodes = order.map((id) => flow.nodes.find((n) => n.id === id)!).filter(Boolean);

    const r = run.steps[lastResponded.id].response!;
    return "body" in r ? r.body : r.bodyJson;
  })();

  return (
    <details className="flow-report" open={run.status !== "running"}>
      <summary>
        Run {run.status}{run.totalMs != null ? ` · ${run.totalMs} ms` : ""} · {nodes.filter((n) => run.steps[n.id]?.status === "success").length}/{nodes.length} steps ok
      </summary>
      <table className="flow-report-table">
        <thead><tr><th /><th>Step</th><th>Status</th><th>Time</th><th>Detail</th></tr></thead>
        <tbody>
          {nodes.map((n) => {
            const r = run.steps[n.id];
            return (
              <tr key={n.id} className={`status-${r?.status ?? "idle"}`} onClick={() => updateFlowTab(tabId, { selectedNodeId: n.id })}>

                <td>{n.key}</td>
                <td>{r?.status ?? "idle"}</td>
                <td>{r?.timeMs != null ? `${r.timeMs} ms` : ""}</td>
                <td className="flow-report-detail">{r?.error ?? (r?.response && "status" in r.response ? `HTTP ${r.response.status}` : r?.response ? r.response.statusCode : "")}</td>
              </tr>
            );
          })}
        </tbody>
      </table>
      {typeof finalBody === "string" && finalBody && (
        <details className="flow-report-final"><summary>Final response ({lastResponded!.key})</summary>
          <pre>{finalBody.length > 20000 ? `${finalBody.slice(0, 20000)}\n…truncated` : finalBody}</pre>
        </details>
